cat <<EOF > simple-webapp-aom.yaml
apiVersion: v1
kind: Pod
metadata:
  name: simple-webapp-aom
  labels:
    name: simple-webapp-aom
spec:
  containers:
  - name: simple-webapp
    image: kodekloud/webapp-delayed-start
    ports:
    - containerPort: 8080
    readinessProbe:
      httpGet:
        path: /ready
        port: 8080
      initialDelaySeconds: 15
      periodSeconds: 5
