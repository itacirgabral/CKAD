> Criar um PersistentVolumeClaim de 100Mi, tipo hostPath no Minikube, e montar em um Pod nginx.
- k run nx --image=nginx --restart=Never --dry-run=client -o yaml
```yaml
apiVersion: v1
kind: Pod
metadata:
  name: nx
spec:
  containers:
  - image: nginx
    name: nx
  restartPolicy: Never
```
----
```yaml
apiVersion: v1
kind: PersistentVolume
metadata:
  name: pv-hostpath
spec:
  capacity:
    storage: 100Mi
  accessModes:
    - ReadWriteOnce
  hostPath:
    path: /tmp/data  # caminho no host Minikube
---
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: pvc-hostpath
spec:
  accessModes:
    - ReadWriteOnce
  resources:
    requests:
      storage: 100Mi
  volumeName: pv-hostpath
```
```yaml
apiVersion: v1
kind: Pod
metadata:
  name: nx
spec:
  containers:
  - name: nx
    image: nginx
    volumeMounts:
    - name: www-storage
      mountPath: /var/www
  volumes:
  - name: www-storage
    persistentVolumeClaim:
      claimName: pvc-hostpath
  restartPolicy: Never
```
